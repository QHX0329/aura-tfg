       ```typescript
       import apiClient from './client';

       export interface OptimizeRequest {
         shopping_list_id: number;
         lat: number;
         lng: number;
         max_distance_km?: number;
         max_stops?: number;
         w_precio?: number;
         w_distancia?: number;
         w_tiempo?: number;
       }

       export interface RouteStop {
         store_id: number;
         store_name: string;
         chain: string;
         lat: number;
         lng: number;
         distance_km: number;
         time_minutes: number;
         products: { product_id: number; name: string; price: number }[];
       }

       export interface OptimizeResponse {
         id: number;
         total_price: number;
         total_distance_km: number;
         estimated_time_minutes: number;
         route: RouteStop[];
       }

       export const optimizeRoute = (data: OptimizeRequest) =>
         apiClient.post<{ success: boolean; data: OptimizeResponse }>('/optimize/', data);
       ```

    2. **ocrService.ts**:
       ```typescript
       import apiClient from './client';

       export interface OCRItem {
         raw_text: string;
         matched_product_id?: number;
         matched_product_name?: string;
         confidence: number;
         quantity: number;
       }

       export interface OCRScanResponse {
         items: OCRItem[];
       }

       export const scanImage = (imageUri: string) => {
         const formData = new FormData();
         formData.append('image', {
           uri: imageUri,
           type: 'image/jpeg',
           name: 'scan.jpg',
         } as any);
         return apiClient.post<{ success: boolean; data: OCRScanResponse }>('/ocr/scan/', formData, {
           headers: { 'Content-Type': 'multipart/form-data' },
         });
       };
       ```

    3. **assistantService.ts**:
       ```typescript
       import apiClient from './client';

       export interface ChatMessage {
         role: 'user' | 'assistant';
         content: string;
       }

       export interface ChatResponse {
         role: 'assistant';
         content: string;
       }

       export const sendChatMessage = (messages: ChatMessage[]) =>
         apiClient.post<{ success: boolean; data: ChatResponse }>('/assistant/chat/', { messages });
       ```
