    ```python
    from django.db import models
    from django.utils.text import slugify

    class Category(models.Model):
        name = models.CharField(max_length=100, verbose_name="Nombre")
        slug = models.SlugField(max_length=120, unique=True, blank=True)
        parent = models.ForeignKey(
            "self", null=True, blank=True, on_delete=models.SET_NULL,
            related_name="children", verbose_name="Categoría padre"
        )
        created_at = models.DateTimeField(auto_now_add=True)

        class Meta:
            verbose_name = "Categoría"
            ordering = ["name"]

        def save(self, *args, **kwargs):
            if not self.slug:
                self.slug = slugify(self.name)
            super().save(*args, **kwargs)

        def clean(self):
            if self.parent and self.parent.parent_id is not None:
                from django.core.exceptions import ValidationError
                raise ValidationError("Máximo 2 niveles de jerarquía permitidos.")


    class Product(models.Model):
        class Unit(models.TextChoices):
            KG = "kg", "Kilogramos"
            G = "g", "Gramos"
            L = "l", "Litros"
            ML = "ml", "Mililitros"
            UNITS = "units", "Unidades"

        name = models.CharField(max_length=200, verbose_name="Nombre")
        normalized_name = models.CharField(max_length=200, blank=True, verbose_name="Nombre normalizado")
        barcode = models.CharField(max_length=20, unique=True, null=True, blank=True, verbose_name="Código de barras")
        category = models.ForeignKey(Category, on_delete=models.SET_NULL, null=True, related_name="products")
        brand = models.CharField(max_length=100, blank=True)
        unit = models.CharField(max_length=10, choices=Unit.choices, default=Unit.UNITS)
        unit_quantity = models.FloatField(default=1.0)
        image_url = models.URLField(blank=True)
        is_active = models.BooleanField(default=True)
        created_at = models.DateTimeField(auto_now_add=True)
        updated_at = models.DateTimeField(auto_now=True)

        class Meta:
            verbose_name = "Producto"
            ordering = ["normalized_name"]

        def save(self, *args, **kwargs):
            if not self.normalized_name:
                self.normalized_name = self.name.lower().strip()
            super().save(*args, **kwargs)


    class ProductProposal(models.Model):
        class Status(models.TextChoices):
            PENDING = "pending", "Pendiente"
            APPROVED = "approved", "Aprobado"
            REJECTED = "rejected", "Rechazado"

        proposed_by = models.ForeignKey(
            "users.User", null=True, blank=True, on_delete=models.SET_NULL, related_name="proposals"
        )
        name = models.CharField(max_length=200)
        brand = models.CharField(max_length=100, blank=True)
        barcode = models.CharField(max_length=20, blank=True)
        category = models.ForeignKey(Category, null=True, blank=True, on_delete=models.SET_NULL)
        image_url = models.URLField(blank=True)
        status = models.CharField(max_length=20, choices=Status.choices, default=Status.PENDING)
        notes = models.TextField(blank=True)
        created_at = models.DateTimeField(auto_now_add=True)

        class Meta:
            verbose_name = "Propuesta de producto"
            ordering = ["-created_at"]
    ```

    - GET /api/v1/products/{id}/ → detail with fields: id, name, normalized_name, barcode, category (id+name), brand, unit, unit_quantity, image_url, price_min, price_max (annotated from Price table — null if no prices)

    - POST /api/v1/products/proposals/ authenticated → 201, status=pending; response includes id and status
    - POST /api/v1/products/proposals/ unauthenticated → 401
    - Trigram search with q="" or q of length < 2 → returns empty list (no DB query)
    - Barcode with unknown value → 404 + {"success": false, "error": {"code": "PRODUCT_NOT_FOUND", ...}}
