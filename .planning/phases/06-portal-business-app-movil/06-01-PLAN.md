```python
"""
Servicios del dominio products.

Funciones:
- approve_proposal: aprueba una propuesta, crea Product y Price en transaccion atomica.
"""

from django.db import transaction
from django.utils import timezone

from apps.products.models import Product, ProductProposal


def approve_proposal(proposal: ProductProposal) -> Product:
    """Aprueba una propuesta: crea/reutiliza Product y materializa Price si procede.

    - Si la propuesta tiene barcode y ya existe un Product con ese barcode,
      reutiliza el producto existente (idempotente).
    - Si la propuesta tiene price y store, crea/actualiza Price con
      source=CROWDSOURCING.
    - Marca la propuesta como APPROVED.

    Args:
        proposal: ProductProposal en estado PENDING.

    Returns:
        Product creado o recuperado.

    Raises:
        IntegrityError: si hay conflicto irrecuperable de clave unica.
    """
    from apps.prices.models import Price

    with transaction.atomic():
        barcode = proposal.barcode or None
        if barcode:
            product, _ = Product.objects.get_or_create(
                barcode=barcode,
                defaults=dict(
                    name=proposal.name,
                    brand=proposal.brand,
                    category=proposal.category,
                    image_url=proposal.image_url,
                    is_active=True,
                ),
            )
        else:
            product = Product.objects.create(
                name=proposal.name,
                brand=proposal.brand,
                barcode=None,
                category=proposal.category,
                image_url=proposal.image_url,
                is_active=True,
            )

        if proposal.price is not None and proposal.store is not None:
            Price.objects.update_or_create(
                product=product,
                store=proposal.store,
                source=Price.Source.CROWDSOURCING,
                defaults={
                    "price": proposal.price,
                    "unit_price": proposal.unit_price,
                    "is_stale": False,
                    "verified_at": timezone.now(),
                },
            )

        proposal.status = ProductProposal.Status.APPROVED
        proposal.save(update_fields=["status"])

    return product
```

The function mirrors the correct logic from `views.py` (transaction.atomic, get_or_create, update_or_create, source=CROWDSOURCING) — NOT the divergent admin.py logic (no transaction, filter/first, create, source=BUSINESS).

```python
def approve(self, request: Request, pk: int | None = None) -> Response:
    """Aprueba una propuesta, crea el producto y materializa el precio si procede."""
    proposal = self.get_object()
    if proposal.status != ProductProposal.Status.PENDING:
        return Response(
            {"detail": "Solo se pueden aprobar propuestas pendientes."},
            status=400,
        )

    from apps.products.serializers import ProductDetailSerializer
    from apps.products.services import approve_proposal

    try:
        product = approve_proposal(proposal)
    except IntegrityError as exc:
        return Response(
            {"detail": f"No se pudo aprobar la propuesta: {exc}"},
            status=409,
        )

    return success_response(
        {"proposal_id": proposal.id, "product": ProductDetailSerializer(product).data}
```

```python
@admin.action(description=_("Aprobar propuestas seleccionadas"))
def approve_proposals(modeladmin, request, queryset):
    """Accion para aprobar propuestas y crear productos reales."""
    from django.db import IntegrityError

    from apps.products.services import approve_proposal

    approved = 0
    skipped = 0
    for proposal in queryset.filter(status="pending"):
        try:
            approve_proposal(proposal)
            approved += 1
        except IntegrityError:
            skipped += 1
            continue

    modeladmin.message_user(
        request,
        f"{approved} propuesta(s) aprobada(s), {skipped} omitida(s) por error.",
    )
```
