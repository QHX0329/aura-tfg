# ── Tests ────────────────────────────────────────────────


class TestBulkUpdate:
    """Tests para POST /api/v1/business/prices/bulk-update/"""

    def test_bulk_update_valid_rows(
        self, api_client, verified_business_user, business_store, product_a, product_b
    ):
        api_client.force_authenticate(user=verified_business_user)
        payload = [
            {"product": product_a.id, "store": business_store.id, "price": "2.50"},
            {"product": product_b.id, "store": business_store.id, "price": "3.10"},
        ]
        response = api_client.post(
            "/api/v1/business/prices/bulk-update/",
            data=payload,
            format="json",
        )
        assert response.status_code == 200
        data = response.json()["data"]
        assert data["created"] == 2
        assert data["updated"] == 0
        assert data["errors"] == []

    def test_bulk_update_invalid_product_id(
        self, api_client, verified_business_user, business_store, product_a
    ):
        api_client.force_authenticate(user=verified_business_user)
        payload = [
            {"product": product_a.id, "store": business_store.id, "price": "2.50"},
            {"product": 9999999, "store": business_store.id, "price": "1.00"},
        ]
        response = api_client.post(
            "/api/v1/business/prices/bulk-update/",
            data=payload,
            format="json",
        )
        assert response.status_code == 200
        data = response.json()["data"]
        assert data["created"] == 1
        assert len(data["errors"]) == 1
        assert data["errors"][0]["index"] == 1
        assert "product" in data["errors"][0]["errors"]

    def test_bulk_update_store_not_owned(
        self, api_client, verified_business_user, other_store, product_a
    ):
        api_client.force_authenticate(user=verified_business_user)
        payload = [
            {"product": product_a.id, "store": other_store.id, "price": "1.50"},
        ]
        response = api_client.post(
            "/api/v1/business/prices/bulk-update/",
            data=payload,
            format="json",
        )
        assert response.status_code == 200
        data = response.json()["data"]
        assert data["created"] == 0
        assert len(data["errors"]) == 1
        assert "Tienda no pertenece" in str(data["errors"][0]["errors"])

    def test_bulk_update_missing_required_field(
        self, api_client, verified_business_user, business_store, product_a
    ):
        api_client.force_authenticate(user=verified_business_user)
        payload = [
            {"product": product_a.id, "store": business_store.id},  # missing price
        ]
        response = api_client.post(
            "/api/v1/business/prices/bulk-update/",
            data=payload,
            format="json",
        )
        assert response.status_code == 200
        data = response.json()["data"]
        assert data["created"] == 0
        assert len(data["errors"]) == 1
        assert "price" in data["errors"][0]["errors"]

    def test_bulk_update_requires_verified_profile(
        self, api_client, unverified_business_user, product_a

        ]
        response = api_client.post(
            "/api/v1/business/prices/bulk-update/",
            data=payload,
            format="json",
        )
        assert response.status_code == 403
