  - Body: {name, brand?, barcode?, category?, image_url?, notes?, price?, unit_price?, store?}
  - Response: 201, {success: true, data: {id, name, status: "pending", ...}}
  - Auth: IsAuthenticated

  - Response: 200, paginated list of proposals
  - Auth: IsAdminUser

POST /api/v1/products/proposals/admin/{id}/approve/ (approve action):
  - Response: 200, {success: true, data: {proposal_id, product: {...}}}
  - Auth: IsAdminUser

POST /api/v1/products/proposals/admin/{id}/reject/ (reject action):
  - Body: {reason?: string}
  - Response: 200, {success: true, data: {proposal_id, status: "rejected"}}
  - Auth: IsAdminUser

POST /api/v1/business/prices/bulk-update/ (bulk_update action):
  - Body: [{product: int, store: int, price: decimal}, ...]
  - Response: 200, {success: true, data: {created: int, updated: int, errors: [...]}}
  - Auth: IsAuthenticated + IsVerifiedBusiness
